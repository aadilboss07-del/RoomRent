// Build advance-usage map: for each friend × month, how much advance is consumed
function buildAdvanceMap(advanceLog, friends, year, yearData) {
  const map = {};
  friends.forEach(friend => {
    let pool = 0;
    const entries = advanceLog
      .filter(e => e.friend===friend)
      .sort((a,b)=>a.date.localeCompare(b.date))
      .map(e=>({...e}));
    map[friend] = {};
    for(let i=0;i<12;i++){
      const monthName = MONTHS[i];
      const yr = parseInt(year);
      entries.forEach(e=>{
        if(!e._done){
          const d=new Date(e.date);
          if(d.getFullYear()<yr||(d.getFullYear()===yr&&d.getMonth()<=i)){
            e._done=true;
            pool+=e.amount;
          }
        }
      });
      const md = yearData[monthName]||{};
      const total=(parseFloat(md.roomRent)||0)+(parseFloat(md.electricity)||0)+(parseFloat(md.other)||0);
      const n=friends.length;
      const base = total>0&&n>0?total/n:0;
      const used = Math.min(pool,base);
      map[friend][monthName]={used,poolBefore:pool,poolAfter:pool-used};
      pool-=used;
    }
  });
  return map;
}

  const setDataP    = fn => setData(prev=>{ const next=typeof fn==="function"?fn(prev):fn; saveLS("rs_data",next); return next; });
  const setAdvLP    = fn => setAdvanceLog(prev=>{ const next=typeof fn==="function"?fn(prev):fn; saveLS("rs_advances",next); return next; });

  const years = [...new Set([String(CURRENT_YEAR-1),String(CURRENT_YEAR),String(CURRENT_YEAR+1),...Object.keys(data)])].sort();
  const yearData = data[selectedYear]||emptyYear(friends);

  const showNotif = (msg,type="success") => {
    setNotif({msg,type});
    clearTimeout(notifTimer.current);
    notifTimer.current=setTimeout(()=>setNotif(null),3200);
  };

  const advanceMap = useMemo(
    ()=>buildAdvanceMap(advanceLog,friends,selectedYear,yearData),
    [advanceLog,friends,selectedYear,yearData]
  );

  const advancePoolRemaining = useMemo(()=>{
    const r={};
    friends.forEach(f=>{
      const total=advanceLog.filter(e=>e.friend===f).reduce((s,e)=>s+e.amount,0);
      const used=Object.values(advanceMap[f]||{}).reduce((s,v)=>s+v.used,0);
      r[f]=total-used;
    });
    return r;
  },[advanceLog,friends,advanceMap]);

  // ── STATS ──
  const getMonthStats = useCallback(month=>{
    const md=yearData[month]||emptyMonth(friends);
    const total=(parseFloat(md.roomRent)||0)+(parseFloat(md.electricity)||0)+(parseFloat(md.other)||0);
    const n=friends.length;
    const base=total>0&&n>0?total/n:0;
    const advUsed={}, adjPH={};
    friends.forEach(f=>{
      const u=advanceMap[f]?.[month]?.used||0;
      advUsed[f]=u; adjPH[f]=Math.max(0,base-u);
    });
    const totalPaid=friends.reduce((s,f)=>s+(parseFloat(md.payments?.[f])||0),0);
    const balances={};
    friends.forEach(f=>{balances[f]=(parseFloat(md.payments?.[f])||0)-adjPH[f];});
    return{total,base,adjPH,advUsed,totalPaid,balances,hasData:total>0||totalPaid>0};
  },[yearData,advanceMap,friends]);

  const getYearSummary = useCallback(()=>{
    let gTotal=0,gPaid=0;
    const fTotals={},fDues={},rows=[];
    friends.forEach(f=>{fTotals[f]=0;fDues[f]=0;});
    MONTHS.forEach(m=>{
      const s=getMonthStats(m);
      gTotal+=s.total; gPaid+=s.totalPaid;
      friends.forEach(f=>{fTotals[f]+=(parseFloat(yearData[m]?.payments?.[f])||0);fDues[f]+=s.balances[f];});
      rows.push({month:m,...s});
    });
    return{gTotal,gPaid,fTotals,fDues,rows};
  },[getMonthStats,yearData,friends]);

  const stats     = getMonthStats(MONTHS[activeMonth]);
  const ys        = getYearSummary();
  const amData    = yearData[MONTHS[activeMonth]]||emptyMonth(friends);

  const fmt = n => n===0?"—":"₹"+Math.abs(n).toLocaleString("en-IN",{maximumFractionDigits:0});
  const fmtDate = d => { const dt=new Date(d); return `${dt.getDate()} ${MONTHS_SHORT[dt.getMonth()]} ${dt.getFullYear()}`; };
  const gcols = n => n<=2?"repeat(2,1fr)":n<=4?"repeat(4,1fr)":n<=6?"repeat(3,1fr)":"repeat(4,1fr)";

  // ── DATA UPDATES ──
  const updateField = (month,field,val) => setDataP(d=>({
    ...d,[selectedYear]:{...(d[selectedYear]||emptyYear(friends)),
      [month]:{...(d[selectedYear]?.[month]||emptyMonth(friends)),[field]:val}}
  }));
  const updatePayment = (month,friend,val) => setDataP(d=>{
    const yd=d[selectedYear]||emptyYear(friends);
    return{...d,[selectedYear]:{...yd,[month]:{...yd[month],payments:{...yd[month]?.payments,[friend]:val}}}};
  });
  const clearMonth = () => {
    setDataP(d=>({...d,[selectedYear]:{...(d[selectedYear]||emptyYear(friends)),[MONTHS[activeMonth]]:emptyMonth(friends)}}));
    showNotif("Month cleared");
  };

  // ── ADVANCE OPS ──
  const openAdvForm = f => setAdvForm(p=>({...p,[f]:{open:true,date:TODAY,amount:"",note:""}}));
  const closeAdvForm = f => setAdvForm(p=>({...p,[f]:{...p[f],open:false}}));
  const saveAdv = f => {
    const form=advForm[f]||{};
    if(!form.amount||isNaN(parseFloat(form.amount))||parseFloat(form.amount)<=0){showNotif("Amount sahi daalo!","error");return;}
    if(!form.date){showNotif("Date zaroori hai!","error");return;}
    const e={id:makeId(),friend:f,date:form.date,amount:parseFloat(form.amount),note:form.note.trim()};
    setAdvLP(p=>[...p,e].sort((a,b)=>a.date.localeCompare(b.date)));
    closeAdvForm(f);
    showNotif(`₹${e.amount.toLocaleString("en-IN")} advance saved for ${f} ✅`);
  };
  const deleteAdv = id => { setAdvLP(p=>p.filter(e=>e.id!==id)); showNotif("Entry deleted"); };

  // ── MEMBER OPS ──
  const addFriend = () => {
    const name=newFriendName.trim();
    if(!name)return;
    if(friends.includes(name)){showNotif("Naam already hai!","error");return;}
    const nf=[...friends,name];
    setFriendsP(nf);
    setDataP(d=>{
      const upd={};
      Object.keys(d).forEach(yr=>{upd[yr]={};
        MONTHS.forEach(m=>upd[yr][m]={...(d[yr][m]||emptyMonth(friends)),payments:{...(d[yr][m]?.payments||{}),[name]:""}});
      });
      return upd;
    });
    setNewFriendName(""); setShowAddForm(false);
    showNotif(`${name} add ho gaya! 🎉`);
  };
  const saveEdit = () => {
    if(!editingFriend)return;
    const{index,value}=editingFriend;
    const newName=value.trim(), oldName=friends[index];
    if(!newName||newName===oldName){setEditingFriend(null);return;}
    if(friends.includes(newName)){showNotif("Naam already hai!","error");return;}
    const nf=[...friends]; nf[index]=newName; setFriendsP(nf);
    setDataP(d=>{
      const upd={};
      Object.keys(d).forEach(yr=>{upd[yr]={};
        MONTHS.forEach(m=>{
          const mo=d[yr][m]||emptyMonth(friends);
          const np={};Object.keys(mo.payments).forEach(k=>{np[k===oldName?newName:k]=mo.payments[k];});
          upd[yr][m]={...mo,payments:np};
        });
      });
      return upd;
    });
    setAdvLP(p=>p.map(e=>e.friend===oldName?{...e,friend:newName}:e));
    setEditingFriend(null);
    showNotif(`"${oldName}" → "${newName}" ✅`);
  };
  const deleteFriend = () => {
    const name=deleteConfirm;
    if(friends.length<=2){showNotif("Min 2 log chahiye!","error");setDeleteConfirm(null);return;}
    setFriendsP(friends.filter(f=>f!==name));
    setDataP(d=>{
      const upd={};
      Object.keys(d).forEach(yr=>{upd[yr]={};
        MONTHS.forEach(m=>{const mo=d[yr][m]||emptyMonth(friends);const np={...mo.payments};delete np[name];upd[yr][m]={...mo,payments:np};});
      });
      return upd;
    });
    setAdvLP(p=>p.filter(e=>e.friend!==name));
    setDeleteConfirm(null); showNotif(`${name} removed`);
  };

  return (
    <div>
      {/* HEADER */}
      <header className="app-header">
        <div className="header-left">
          <div className="app-title">Room<span>Split</span> <span style={{fontSize:"1rem",color:"var(--text3)"}}>Ultra</span></div>
          <div className="app-sub">{friends.length} ROOMMATES · EXPENSE TRACKER</div>
        </div>
        <select className="year-select" value={selectedYear} onChange={e=>setSelectedYear(e.target.value)}>
          {years.map(y=><option key={y} value={y}>{y}</option>)}
        </select>
      </header>

      {/* TABS */}
      <div className="tabs">
        {[["entry","📝","Entry"],["summary","📊","Summary"],["advances","💳","Advances"],["members","👥","Members"]].map(([id,icon,label])=>(
          <button key={id} className={`tab-btn ${activeTab===id?"active":""}`} onClick={()=>setActiveTab(id)}>
            <span>{icon}</span>{label}
            {id==="advances"&&advanceLog.length>0&&<span className="tab-badge">{advanceLog.length}</span>}
          </button>
        ))}
      </div>

      {/* ── ENTRY TAB ── */}
      {activeTab==="entry"&&(
        <>
          <div className="month-strip-wrap">
            <div className="month-strip">
              {MONTHS_SHORT.map((m,i)=>{
                const ms=getMonthStats(MONTHS[i]);
                return <button key={m} className={`month-chip${activeMonth===i?" active":""}${ms.hasData?" has-data":""}`} onClick={()=>setActiveMonth(i)}>{m}</button>;
              })}
            </div>
          </div>

          <div className="month-section-title">{MONTHS[activeMonth]} <span>{selectedYear}</span></div>

          <div className="card">
            <div className="card-header">
              <div className="card-title"><div className="card-title-icon">🏠</div>Expenses</div>
              <button className="clear-btn" onClick={clearMonth}>Clear Month</button>
            </div>
            <div className="expense-grid">
              {[["roomRent","Room Rent","e.g. 24000"],["electricity","Electricity","e.g. 1800"],["other","Other","e.g. 500"]].map(([key,lbl,ph])=>(
                <div className="expense-field" key={key}>
                  <label className="field-label">{lbl}</label>
                  <input className="field-input" type="number" placeholder={ph} value={amData[key]||""} onChange={e=>updateField(MONTHS[activeMonth],key,e.target.value)}/>
                </div>
              ))}
            </div>
            <div className="stats-row">
              <div className="stat-box highlight"><div className="stat-val">{fmt(stats.total)}</div><div className="stat-lbl">Total</div></div>
              <div className="stat-box"><div className="stat-val">{fmt(stats.base)}</div><div className="stat-lbl">Base /Head</div></div>
              <div className="stat-box"><div className="stat-val">{fmt(stats.totalPaid)}</div><div className="stat-lbl">Paid</div></div>
              <div className="stat-box">
                <div className="stat-val" style={{color:stats.total-stats.totalPaid<=0?"var(--green)":"var(--red)"}}>{fmt(Math.abs(stats.total-stats.totalPaid))}</div>
                <div className="stat-lbl">{stats.total-stats.totalPaid<=0?"Surplus":"Remaining"}</div>
              </div>
            </div>
            {stats.total>0&&(
              <div className="per-head-row" style={{gridTemplateColumns:gcols(friends.length)}}>
                {friends.map((f,idx)=>{
                  const c=gc(idx); const used=stats.advUsed[f]||0;
                  return (
                    <div className="per-head-item" key={f} style={{borderColor:c.border,background:c.bg}}>
                      <div className="per-head-friend" style={{color:c.color}}>{f}</div>
                      <div className="per-head-amt">{fmt(stats.adjPH[f])}</div>
                      {used>0&&<div className="per-head-adv">✓ Adv: ₹{used.toLocaleString("en-IN")}</div>}
                    </div>
                  );
                })}
              </div>
            )}
          </div>

          <div className="card">
            <div className="card-header">
              <div className="card-title"><div className="card-title-icon">💳</div>Payments Made</div>
            </div>
            <div className="payments-grid" style={{gridTemplateColumns:gcols(friends.length)}}>
              {friends.map((f,idx)=>{
                const c=gc(idx);
                const bal=stats.balances[f];
                const bClass=bal>0.5?"balance-pos":bal<-0.5?"balance-neg":"balance-zero";
                const bText=bal>0.5?`+${fmt(bal)}`:bal<-0.5?`-${fmt(bal)}`:"Even";
                const pool=advancePoolRemaining[f]||0;
                return (
                  <div className="payment-card" key={f} style={{borderColor:c.border,background:c.bg}}>
                    <div className="friend-name" style={{color:c.color}}>{f}</div>
                    <input className="payment-input" type="number" placeholder="₹ Paid" value={amData.payments?.[f]??""} onChange={e=>updatePayment(MONTHS[activeMonth],f,e.target.value)}/>
                    <div className={`balance-badge ${bClass}`}>{stats.total>0?bText:"—"}</div>
                    {pool>0&&<div className="pool-badge">Pool: ₹{pool.toLocaleString("en-IN")} left</div>}
                  </div>
                );
              })}
            </div>
          </div>
        </>
      )}

      {/* ── SUMMARY TAB ── */}
      {activeTab==="summary"&&(
        <>
          <div className="summary-total-card">
            <div>
              <div className="total-amount">₹{ys.gTotal.toLocaleString("en-IN",{maximumFractionDigits:0})}</div>
              <div className="total-label">Total Expenses · {selectedYear}</div>
            </div>
            <div style={{textAlign:"right"}}>
              <div style={{fontFamily:"'JetBrains Mono',monospace",fontSize:"1.3rem",fontWeight:600,color:"var(--text)"}}>₹{ys.gPaid.toLocaleString("en-IN",{maximumFractionDigits:0})}</div>
              <div className="total-label">Total Collected</div>
            </div>
          </div>

          <div className="friend-summary-grid">
            {friends.map((f,idx)=>{
              const c=gc(idx);
              const totalAdv=advanceLog.filter(e=>e.friend===f).reduce((s,e)=>s+e.amount,0);
              return (
                <div className="friend-sum-card" key={f} style={{background:c.bg,borderColor:c.border}}>
                  <div className="friend-sum-icon" style={{background:c.color+"22",color:c.color}}>{f[0]}</div>
                  <div className="friend-sum-info">
                    <div className="friend-sum-name" style={{color:c.color}}>{f}</div>
                    <div className="friend-sum-row">
                      <div>
                        <div style={{fontSize:"0.62rem",color:"var(--text3)",fontWeight:600,textTransform:"uppercase",letterSpacing:"0.06em",marginBottom:2}}>Paid</div>
                        <div className="friend-sum-paid">₹{ys.fTotals[f].toLocaleString("en-IN",{maximumFractionDigits:0})}</div>
                      </div>
                      <div style={{textAlign:"right"}}>
                        <div style={{fontSize:"0.62rem",color:"var(--text3)",fontWeight:600,textTransform:"uppercase",letterSpacing:"0.06em",marginBottom:2}}>Balance</div>
                        <div className={ys.fDues[f]>=-0.5?"due-pos":"due-neg"}>
                          {Math.abs(ys.fDues[f])<0.5?"Even":ys.fDues[f]>0?`+₹${ys.fDues[f].toLocaleString("en-IN",{maximumFractionDigits:0})}`:"-₹"+Math.abs(ys.fDues[f]).toLocaleString("en-IN",{maximumFractionDigits:0})}
                        </div>
                      </div>
                    </div>
                    {totalAdv>0&&<div style={{marginTop:5,fontSize:"0.62rem",color:"var(--green)",fontFamily:"'JetBrains Mono',monospace",fontWeight:600}}>💰 Advance: ₹{totalAdv.toLocaleString("en-IN")}</div>}
                  </div>
                </div>
              );
            })}
          </div>

          <div className="card" style={{overflowX:"auto"}}>
            <div className="card-header">
              <div className="card-title"><div className="card-title-icon">📅</div>Month-wise Breakdown</div>
            </div>
            <table className="month-table">
              <thead>
                <tr><th>Month</th><th>Total</th>{friends.map(f=><th key={f}>{f.slice(0,3)}</th>)}<th>Bal</th></tr>
              </thead>
              <tbody>
                {ys.rows.map(({month,total,totalPaid,hasData})=>(
                  <tr key={month} className={!hasData?"empty-row":""}>
                    <td>{month.slice(0,3)}</td>
                    <td>{total>0?`₹${total.toLocaleString("en-IN",{maximumFractionDigits:0})}`:"—"}</td>
                    {friends.map(f=>{
                      const paid=parseFloat(yearData[month]?.payments?.[f])||0;
                      const advU=advanceMap[f]?.[month]?.used||0;
                      return <td key={f} style={{color:paid>0||advU>0?"var(--text)":"var(--text3)"}}>
                        {paid>0?`₹${paid.toLocaleString("en-IN",{maximumFractionDigits:0})}`:advU>0?<span style={{color:"var(--green)",fontSize:"0.75rem"}}>adv</span>:"—"}
                      </td>;
                    })}
                    <td style={{color:total===0?"var(--text3)":totalPaid>=total?"var(--green)":"var(--red)"}}>
                      {total>0?(totalPaid>=total?`+₹${(totalPaid-total).toLocaleString("en-IN",{maximumFractionDigits:0})}`:"-₹"+(total-totalPaid).toLocaleString("en-IN",{maximumFractionDigits:0})):"—"}
                    </td>
                  </tr>
                ))}
              </tbody>
            </table>
          </div>
        </>
      )}

      {/* ── ADVANCES TAB ── */}
      {activeTab==="advances"&&(
        <>
          {/* Pool summary */}
          <div className="card" style={{marginBottom:16}}>
            <div className="card-header" style={{marginBottom:10}}>
              <div className="card-title"><div className="card-title-icon">💰</div>Advance Pool Summary</div>
            </div>
            <div style={{display:"grid",gridTemplateColumns:gcols(friends.length),gap:10}}>
              {friends.map((f,idx)=>{
                const c=gc(idx);
                const total=advanceLog.filter(e=>e.friend===f).reduce((s,e)=>s+e.amount,0);
                const used=Object.values(advanceMap[f]||{}).reduce((s,v)=>s+v.used,0);
                const rem=total-used;
                return (
                  <div key={f} style={{background:c.bg,border:`1.5px solid ${c.border}`,borderRadius:10,padding:"12px 14px"}}>
                    <div style={{fontSize:"0.72rem",fontWeight:700,color:c.color,textTransform:"uppercase",letterSpacing:"0.06em",marginBottom:6}}>{f}</div>
                    <div style={{fontFamily:"'JetBrains Mono',monospace",fontSize:"1rem",fontWeight:600,color:"var(--text)"}}>₹{total.toLocaleString("en-IN")}</div>
                    <div style={{fontSize:"0.65rem",color:"var(--text3)",marginTop:3}}>Total given</div>
                    {used>0&&<div style={{fontSize:"0.68rem",color:"var(--green)",fontFamily:"'JetBrains Mono',monospace",fontWeight:600,marginTop:4}}>✓ Used: ₹{used.toLocaleString("en-IN")}</div>}
                    {rem>0&&<div style={{fontSize:"0.68rem",color:"var(--accent)",fontFamily:"'JetBrains Mono',monospace",fontWeight:600,marginTop:2}}>⏳ Left: ₹{rem.toLocaleString("en-IN")}</div>}
                  </div>
                );
              })}
            </div>
          </div>

          {/* Per-friend log */}
          {friends.map((f,idx)=>{
            const c=gc(idx);
            const entries=advanceLog.filter(e=>e.friend===f).sort((a,b)=>a.date.localeCompare(b.date));
            const form=advForm[f]||{};
            const total=entries.reduce((s,e)=>s+e.amount,0);
            const rem=advancePoolRemaining[f]||0;
            return (
              <div className="card" key={f}>
                <div style={{display:"flex",alignItems:"center",justifyContent:"space-between",marginBottom:14,flexWrap:"wrap",gap:8}}>
                  <div style={{display:"flex",alignItems:"center",gap:10}}>
                    <div style={{width:36,height:36,borderRadius:"50%",background:c.color+"22",color:c.color,display:"flex",alignItems:"center",justifyContent:"center",fontWeight:800,fontSize:"1rem",flexShrink:0}}>{f[0]}</div>
                    <div>
                      <div style={{fontWeight:800,fontSize:"0.92rem",color:c.color}}>{f}</div>
                      <div style={{fontSize:"0.68rem",color:"var(--text3)",fontFamily:"'JetBrains Mono',monospace",marginTop:1}}>
                        {entries.length} entr{entries.length===1?"y":"ies"} · Total: ₹{total.toLocaleString("en-IN")}
                      </div>
                    </div>
                  </div>
                  <div style={{display:"flex",gap:8,alignItems:"center",flexWrap:"wrap"}}>
                    {rem>0&&<div style={{fontFamily:"'JetBrains Mono',monospace",fontSize:"0.75rem",fontWeight:700,padding:"4px 12px",borderRadius:999,background:"var(--green-bg)",color:"var(--green)",border:"1px solid #10b98130"}}>⏳ ₹{rem.toLocaleString("en-IN")} pending</div>}
                    {!form.open&&<button className="add-adv-trigger" onClick={()=>openAdvForm(f)}>＋ Advance Add</button>}
                  </div>
                </div>

                {entries.length===0&&!form.open
                  ?<div className="adv-empty">Koi advance entry nahi hai</div>
                  :<div className="adv-log-list">
                    {entries.map(e=>(
                      <div className="adv-entry-row" key={e.id}>
                        <div className="adv-entry-date">📅 {fmtDate(e.date)}</div>
                        <div className="adv-entry-amt">+₹{e.amount.toLocaleString("en-IN")}</div>
                        <div className="adv-entry-note">{e.note||<span style={{color:"var(--text3)",fontStyle:"normal",opacity:.5}}>No note</span>}</div>
                        <button className="adv-entry-del" onClick={()=>deleteAdv(e.id)}>🗑</button>
                      </div>
                    ))}
                  </div>
                }

                {form.open&&(
                  <div className="add-adv-form">
                    <div style={{fontSize:"0.72rem",fontWeight:700,color:"var(--accent)",textTransform:"uppercase",letterSpacing:"0.08em"}}>Naya Advance Entry</div>
                    <div className="add-adv-row">
                      <div>
                        <div className="field-label" style={{marginBottom:5}}>Date</div>
                        <input type="date" className="adv-input" value={form.date||TODAY} max={TODAY}
                          onChange={e=>setAdvForm(p=>({...p,[f]:{...p[f],date:e.target.value}}))}/>
                      </div>
                      <div>
                        <div className="field-label" style={{marginBottom:5}}>Amount (₹)</div>
                        <input type="number" className="adv-input" placeholder="e.g. 2000" value={form.amount||""}
                          onChange={e=>setAdvForm(p=>({...p,[f]:{...p[f],amount:e.target.value}}))}/>
                      </div>
                    </div>
                    <div>
                      <div className="field-label" style={{marginBottom:5}}>Note (optional)</div>
                      <input type="text" className="adv-note-input" placeholder="e.g. March ka advance, February me diya tha..."
                        value={form.note||""}
                        onChange={e=>setAdvForm(p=>({...p,[f]:{...p[f],note:e.target.value}}))}
                        onKeyDown={e=>{if(e.key==="Enter")saveAdv(f);if(e.key==="Escape")closeAdvForm(f);}}/>
                    </div>
                    <div className="adv-form-btns">
                      <button className="adv-cancel-btn" onClick={()=>closeAdvForm(f)}>Cancel</button>
                      <button className="adv-save-btn" onClick={()=>saveAdv(f)} disabled={!form.amount}>Save ✓</button>
                    </div>
                  </div>
                )}

                {entries.length>0&&(
                  <>
                    <div className="divider"/>
                    <div className="section-label">Auto-Adjust — {selectedYear}</div>
                    <div style={{display:"flex",flexWrap:"wrap",gap:6}}>
                      {MONTHS.map((m,i)=>{
                        const used=advanceMap[f]?.[m]?.used||0;
                        const pb=advanceMap[f]?.[m]?.poolBefore||0;
                        if(pb===0&&used===0)return null;
                        return (
                          <div key={m} style={{display:"flex",alignItems:"center",gap:5,padding:"4px 10px",borderRadius:999,background:used>0?"var(--green-bg)":"var(--surface2)",border:`1px solid ${used>0?"#10b98130":"var(--border)"}`,fontSize:"0.7rem",fontFamily:"'JetBrains Mono',monospace"}}>
                            <span style={{color:"var(--text3)",fontFamily:"'Syne',sans-serif",fontWeight:600}}>{MONTHS_SHORT[i]}</span>
                            {used>0?<span style={{color:"var(--green)",fontWeight:600}}>−₹{used.toLocaleString("en-IN")}</span>:<span style={{color:"var(--text3)"}}>—</span>}
                          </div>
                        );
                      })}
                    </div>
                  </>
                )}
              </div>
            );
          })}
        </>
      )}

      {/* ── MEMBERS TAB ── */}
      {activeTab==="members"&&(
        <div className="card">
          <div className="card-header">
            <div className="card-title"><div className="card-title-icon">👥</div>Manage Roommates</div>
            <div className="info-pill">{friends.length} members</div>
          </div>
          <p style={{fontSize:"0.78rem",color:"var(--text3)",marginBottom:16,lineHeight:1.5}}>
            ✏️ Naam edit · ➕ Naya add · 🗑️ Remove — saara data update ho jaata hai
          </p>
          {friends.map((f,idx)=>{
            const c=gc(idx);
            const isEditing=editingFriend?.index===idx;
            const advCount=advanceLog.filter(e=>e.friend===f).length;
            return (
              <div className="friend-row" key={f} style={{borderColor:isEditing?"var(--accent-border)":""}}>
                <div className="color-dot" style={{background:c.color}}/>
                {isEditing?(
                  <>
                    <input className="friend-edit-input" value={editingFriend.value}
                      onChange={e=>setEditingFriend({...editingFriend,value:e.target.value})}
                      onKeyDown={e=>{if(e.key==="Enter")saveEdit();if(e.key==="Escape")setEditingFriend(null);}}
                      autoFocus/>
                    <div className="row-actions">
                      <button className="save-btn" onClick={saveEdit}>Save ✓</button>
                      <button className="cancel-btn" onClick={()=>setEditingFriend(null)}>✕</button>
                    </div>
                  </>
                ):(
                  <>
                    <div className="friend-row-name" style={{color:c.color}}>{f}</div>
                    {advCount>0&&<div style={{fontSize:"0.65rem",color:"var(--green)",fontWeight:600,fontFamily:"'JetBrains Mono',monospace"}}>💰{advCount}</div>}
                    <div className="row-actions">
                      <button className="icon-btn" onClick={()=>setEditingFriend({index:idx,value:f})}>✏️</button>
                      {friends.length>2&&<button className="icon-btn danger" onClick={()=>setDeleteConfirm(f)}>🗑️</button>}
                    </div>
                  </>
                )}
              </div>
            );
          })}
          {showAddForm?(
            <div className="add-friend-form">
              <input className="add-friend-input" placeholder="Naya roommate ka naam..." value={newFriendName}
                onChange={e=>setNewFriendName(e.target.value)}
                onKeyDown={e=>{if(e.key==="Enter")addFriend();if(e.key==="Escape"){setShowAddForm(false);setNewFriendName("");}}}
                autoFocus/>
              <button className="confirm-add-btn" onClick={addFriend} disabled={!newFriendName.trim()}>Add ✓</button>
              <button className="cancel-btn" style={{padding:"10px 12px"}} onClick={()=>{setShowAddForm(false);setNewFriendName("");}}>✕</button>
            </div>
          ):(
            <button className="add-friend-btn" onClick={()=>setShowAddForm(true)}>＋ Naya Roommate Add Karo</button>
          )}
        </div>
      )}

      {/* DELETE MODAL */}
      {deleteConfirm&&(
        <div className="modal-overlay" onClick={()=>setDeleteConfirm(null)}>
          <div className="modal" onClick={e=>e.stopPropagation()}>
            <div className="modal-title">"{deleteConfirm}" ko remove karo?</div>
            <div className="modal-sub">Iska saara data aur advance history bhi delete ho jayegi. Undo nahi hoga.</div>
            <div className="modal-actions">
              <button className="modal-cancel-btn" onClick={()=>setDeleteConfirm(null)}>Rehne Do</button>
              <button className="modal-danger-btn" onClick={deleteFriend}>Haan, Remove</button>
            </div>
          </div>
        </div>
      )}

      {notif&&<div className={`notif notif-${notif.type}`}>{notif.msg}</div>}
    </div>
  );
}

const root = ReactDOM.createRoot(document.getElementById("root"));
root.render(<App />);
